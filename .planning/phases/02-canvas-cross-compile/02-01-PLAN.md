**Locked decisions (per CONTEXT.md and RESEARCH.md):**
- Node A: cyan `#00DDFF` → `vec4(0.0, 0.867, 1.0, 1.0)`
- Node B: magenta `#FF44AA` → `vec4(1.0, 0.267, 0.667, 1.0)`
- Node radius: 25.0 px (diam = 50x50)
- Starting positions: Node A at (0.30, 0.40), Node B at (0.70, 0.60)
- Canvas background: `#111118`
- Only one node draggable at a time; Node A checked first (higher priority)
- Positions clamped to canvas bounds [0.0, 1.0] normalized
- `cx.redraw_all()` on FingerMove (correctness over performance for Phase 2)

**Full file structure to implement:**

```rust
use makepad_widgets::*;
use crate::nodes::NodeState;

live_design! {
    use link::shaders::*;

    DrawNode = {{DrawNode}} {
        instance node_color: vec4

        fn pixel(self) -> vec4 {
            let sdf = Sdf2d::viewport(self.pos * self.rect_size);
            sdf.circle(
                self.rect_size.x * 0.5,
                self.rect_size.y * 0.5,
                self.rect_size.x * 0.5 - 1.5
            );
            sdf.fill(self.node_color);
            return sdf.result;
        }
    }

    pub CanvasWidget = {{CanvasWidget}} {
        width: Fill, height: Fill
        draw_bg: { color: #111118 }
    }
}

#[derive(Live, LiveHook)]
pub struct DrawNode {
    #[deref] draw_vars: DrawVars,
    #[live] pub node_color: Vec4,
}

impl LiveRegister for DrawNode {}

#[derive(Live, LiveHook, Widget)]
pub struct CanvasWidget {
    #[walk]   walk: Walk,
    #[layout] layout: Layout,
    #[live]   draw_bg: DrawQuad,
    #[live]   draw_node_a: DrawNode,
    #[live]   draw_node_b: DrawNode,
    #[rust]   nodes: [NodeState; 2],
    #[rust]   active_node: Option<usize>,
    #[rust]   canvas_rect: Rect,
}

impl CanvasWidget {
    fn update_node_pos(&mut self, idx: usize, abs: DVec2) {
        let r = &self.canvas_rect;
        let nx = ((abs.x - r.pos.x) / r.size.x).clamp(0.0, 1.0) as f32;
        let ny = ((abs.y - r.pos.y) / r.size.y).clamp(0.0, 1.0) as f32;
        self.nodes[idx].x = nx;
        self.nodes[idx].y = ny;
    }
}

impl Widget for CanvasWidget {
    fn handle_event(&mut self, cx: &mut Cx, event: &Event, scope: &mut Scope) {
        // Node A — checked first (higher priority when overlapping)
        match event.hits(cx, self.draw_node_a.area()) {
            Hit::FingerDown(fd) if fd.device.is_primary_hit() => {
                self.active_node = Some(0);
            }
            Hit::FingerMove(fm) if self.active_node == Some(0) => {
                self.update_node_pos(0, fm.abs);
                cx.redraw_all();
            }
            Hit::FingerUp(_) if self.active_node == Some(0) => {
                self.active_node = None;
            }
            _ => {}
        }
        // Node B — checked second
        match event.hits(cx, self.draw_node_b.area()) {
            Hit::FingerDown(fd) if fd.device.is_primary_hit() => {
                if self.active_node.is_none() {
                    self.active_node = Some(1);
                }
            }
            Hit::FingerMove(fm) if self.active_node == Some(1) => {
                self.update_node_pos(1, fm.abs);
                cx.redraw_all();
            }
            Hit::FingerUp(_) if self.active_node == Some(1) => {
                self.active_node = None;
            }
            _ => {}
        }
    }

    fn draw_walk(&mut self, cx: &mut Cx2d, _scope: &mut Scope, walk: Walk) -> DrawStep {
        let rect = cx.walk_turtle(walk);
        self.canvas_rect = rect;

        // Draw background
        self.draw_bg.draw_abs(cx, rect);

        const R: f64 = 25.0;
        let diam = dvec2(R * 2.0, R * 2.0);

        // Node A — cyan, starting at (0.30, 0.40)
        let pos_a = dvec2(
            rect.pos.x + self.nodes[0].x as f64 * rect.size.x - R,
            rect.pos.y + self.nodes[0].y as f64 * rect.size.y - R,
        );
        self.draw_node_a.node_color = Vec4::new(0.0, 0.867, 1.0, 1.0);
        self.draw_node_a.draw_abs(cx, Rect { pos: pos_a, size: diam });

        // Node B — magenta, starting at (0.70, 0.60)
        let pos_b = dvec2(
            rect.pos.x + self.nodes[1].x as f64 * rect.size.x - R,
            rect.pos.y + self.nodes[1].y as f64 * rect.size.y - R,
        );
        self.draw_node_b.node_color = Vec4::new(1.0, 0.267, 0.667, 1.0);
        self.draw_node_b.draw_abs(cx, Rect { pos: pos_b, size: diam });

        DrawStep::done()
    }
}

impl LiveHook for CanvasWidget {
    fn after_new_from_doc(&mut self, _cx: &mut Cx) {
        self.nodes = [
            NodeState::initial_pos(0.30, 0.40, 440.0),
            NodeState::initial_pos(0.70, 0.60, 660.0),
        ];
    }
}
```

- Pitfall 4: `instance node_color` (NOT `uniform`) so each node can be a different color.
- Pitfall 5: Gate FingerMove on `active_node == Some(idx)`.
- Pitfall 3: Always subtract `canvas_rect.pos` when normalizing coords in `update_node_pos`.
