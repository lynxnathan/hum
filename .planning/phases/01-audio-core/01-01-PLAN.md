    **build_stream** — allocation-free callback, center pan (pan_a/pan_b read but Phase 1 ignores them — center mix only):
    ```rust
    pub fn build_stream(
        device: cpal::Device,
        config: cpal::SupportedStreamConfig,
        mut graph_a: Box<dyn AudioUnit + Send>,
        mut graph_b: Box<dyn AudioUnit + Send>,
    ) -> cpal::Stream {
        let stream_config: cpal::StreamConfig = config.into();

        device
            .build_output_stream(
                &stream_config,
                move |data: &mut [f32], _: &cpal::OutputCallbackInfo| {
                    let frames = data.len() / 2;
                    for i in 0..frames {
                        // Phase 1: center pan — equal mix both channels
                        // No Vec, String, Box::new, or Arc::clone in this closure
                        let sa = graph_a.get_mono() * 0.4;
                        let sb = graph_b.get_mono() * 0.4;
                        data[i * 2] = sa + sb;         // left
                        data[i * 2 + 1] = sa + sb;    // right
                    }
                },
                |err| eprintln!("audio stream error: {err}"),
                None,
            )
            .expect("failed to build output stream")
    }
    ```

    **init_audio_async** — device discovery on spawned thread, stream built on caller (main) thread:
    ```rust
    pub fn init_audio_async() -> (cpal::Stream, AudioParams) {
        let (tx, rx) = bounded(1);

        std::thread::spawn(move || {
            let host = cpal::default_host();
            let device = host
                .default_output_device()
                .expect("no output device found");
            let config = device
                .default_output_config()
                .expect("could not get default output config");
            tx.send((device, config)).unwrap();
        });

        let (device, config) = rx.recv().unwrap();
        let sample_rate = config.sample_rate().0;

        let params = AudioParams::new();
        let (graph_a, graph_b) = build_graph(&params, sample_rate);
        let stream = build_stream(device, config, graph_a, graph_b);
        stream.play().expect("failed to start audio stream");
        (stream, params)
    }
    ```
